-받아들일시 bMinusFlag = 1;





if (vec[i] == 위 연산자)       ('여는' 괄호, 덧셈, 뺄셈, 곱셈, 나눗셈)
{
	bCanMinusFlag = 1;
}


if (bCanMinusFlag && vec[i] == '-')
{
	bMinusFlag = 1;
}
